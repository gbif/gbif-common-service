016package org.gbif.service.guice;
017
018import java.util.Map;
019import java.util.Properties;
020
021import com.google.common.collect.ImmutableMap;
022import com.google.common.collect.Maps;
023import com.google.inject.PrivateModule;
024import com.google.inject.name.Names;
025
026/**
027 * Abstract Guice Module to use for private service modules that depend on properties for configuration.
028 * To use this instantiate this module using the constructors and provide the required properties.
029 * All properties have to be prefixed with a custom, shared portion,
030 * for example {@code checklistbank.db} as in {@code checklistbank.db.JDBC.driver}.
031 * All properties will automatically be available as named strings in guice bindings.
032 * To add custom bindings please use configureService instead of overriding configure().
033 *
034 * @see #configureService()
035 */
036
037public abstract class PrivateServiceModule extends PrivateModule {
038
039  private final String propertyPrefix;
040  private final Map<String, String> properties;
041
042  // This is needed for some scenarios which need to install other modules relying on
043  // the properties, but using different prefixes
044  private final Properties verbatimProperties;
045
046  /**
047   * Uses the given properties to configure the service.
048   *
049   * @param properties to use
050   */
051  public PrivateServiceModule(String propertyPrefix, Properties properties) {
052    this.propertyPrefix = propertyPrefix;
053    this.verbatimProperties = properties;
054    this.properties = buildProperties(properties);
055  }
056
057  /**
058   * Filters out all properties that don't have a common prefix and removes that prefix for those properties that do
059   * have it.
060   *
061   * @param properties to filter and translate
062   *
063   * @return immutable Map with the translated and filtered properties
064   */
065  private Map<String, String> buildProperties(Properties properties) {
066    ImmutableMap.Builder<String, String> builder = ImmutableMap.builder();
067
068    for (Map.Entry<String, String> entry : Maps.fromProperties(properties).entrySet()) {
069      if (entry.getKey().startsWith(propertyPrefix)) {
070        builder.put(entry.getKey().substring(propertyPrefix.length()), entry.getValue());
071      }
072    }
073
074    return builder.build();
075  }
076
077  @Override
078  protected final void configure() {
079    Names.bindProperties(binder(), properties);
080    configureService();
081  }
082
083  /**
084   * Implement this method to configure the guice module.
085   */
086  protected abstract void configureService();
087
088  /**
089   * Only subclasses are intended to use this.
090   *
091   * @return The verbatim properties that the module was instantiated with (before trimming prefixes)
092   */
093  protected Properties getVerbatimProperties() {
094    return verbatimProperties;
095  }
096}
